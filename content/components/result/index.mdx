Result 用于反馈操作结果，支持成功、失败、警告等状态。

<ComponentShowcase id="result" path="result/basic" />

```tsx
import { Result } from "@/components/ui/result"
import { Button } from "@/components/ui/button"
import { CheckCircle } from "lucide-react"
export default function Demo() {
  return (
    <Result icon={<CheckCircle className="size-12 text-green-500" />} title="操作成功" description="预计 5 分钟后生效。">
      <Button color="primary" variant="solid">返回首页</Button>
    </Result>
  )
}
```

| description | 描述 | `ReactNode` |
