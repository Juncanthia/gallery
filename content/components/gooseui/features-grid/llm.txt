---
component: FeaturesGrid
slug: gooseui/features-grid

import: |
  import { FeaturesGrid } from "@/components/ui/features-grid"

usage: |
  <FeaturesGrid
    title="为什么选择我们"
    subtitle="专业的开发工具，助你高效构建现代化产品"
  />

props:
  FeaturesGrid:
    title?: string  -- 区块主标题，默认 "Everything for Modern Development"
    subtitle?: string  -- 区块副标题/描述，默认 "Components that save time and help you build quality products"
    features?: Feature[]  -- 特性数据数组，默认 6 项内置示例
    className?: string  -- 附加 CSS 类名
  Feature:
    icon: React.ComponentType<{ className?: string }>  -- 图标组件（必填）
    title: string  -- 特性标题（必填）
    description: string  -- 特性描述（必填）

constraints:
  - 响应式 CSS Grid 布局：移动端 1 列、平板 2 列、桌面 3 列。
  - 每个 FeatureCard 带有 hover 状态（hover:bg-muted/50）。
  - features 数组中的 icon 必须是 React 组件，不能是字符串或 URL。
  - 不传 props 时展示 6 项内置默认特性。
  - 组件为营销/展示型区块，不应在卡片内嵌入表单或交互控件。
  - 不要引入 Ant Design 或 antd 相关依赖。
