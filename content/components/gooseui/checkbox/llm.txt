---
component: Checkbox
slug: gooseui/checkbox

import: import { Checkbox } from "@/components/ui/checkbox"

usage: |
  <Checkbox label="同意条款" />
  <Checkbox label="已选中" defaultChecked />
  <Checkbox label="禁用" disabled />
  <Checkbox label="危险" variant="destructive" />

props:
  variant?: "default" | "destructive" | "success"  -- 颜色变体，默认 "default"
  size?: "sm" | "default" | "lg"  -- 尺寸，默认 "default"
  label?: string  -- 文字标签
  id?: string  -- 输入 id，未传则自动生成
  className?: string  -- 附加样式类名
  checked?: boolean  -- 受控选中状态
  defaultChecked?: boolean  -- 非受控初始选中状态
  disabled?: boolean  -- 禁用
  onChange?: (e: ChangeEvent<HTMLInputElement>) => void  -- 选中变化回调
  required?: boolean  -- 必填
  name?: string  -- 表单字段名
  value?: string  -- 表单值

constraints:
  - 基于原生 <input type="checkbox"> + SVG path 动画实现。
  - 勾选形态由 stroke-dasharray / stroke-dashoffset 动画实现，不依赖外部图标库。
  - 样式定制通过 CSS 变量 --checkbox-checked、--checkbox-unchecked、--checkbox-size 控制。
  - 不包含表单验证逻辑；验证由父级 Form 组件处理。
  - 不要引入 Ant Design Checkbox 或 rc-checkbox。
  - 不要引入 lucide-react 等图标库用于勾选图标。

example: content/components/gooseui/checkbox/examples/basic.tsx
docs: content/components/gooseui/checkbox/index.mdx
