import: |
  import { FAQAccordion } from "@/components/marketing-blocks/gooseui/faq/faq-accordion"

usage: |
  <FAQAccordion />

exports:
  FAQAccordion: 手风琴式 FAQ 组件，以序号编号+展开动画展示问答列表
  FAQItem: 问答项类型 { question: string, answer: string }
  FAQAccordionProps: 组件 Props 类型 { items?: FAQItem[], className?: string }

props:
  FAQAccordion:
    items?: { question: string, answer: string }[]  -- 问答列表，默认内置一组 GooseUI 相关示例
    className?: string  -- 附加样式类名

constraints:
  - 从 @/components/marketing-blocks/gooseui/faq/faq-accordion 导入，不要从 gooseui 内部路径导入。
  - 单展开手风琴模式，每次仅一项展开，再次点击已展开项收起全部。
  - 使用 CSS Grid grid-rows-[1fr]/[0fr] 实现高度展开/收起动画。
  - Plus 图标旋转 45 度表示展开状态，序号数字颜色变化辅助视觉反馈。
  - 无自定义数据时直接 <FAQAccordion /> 使用默认数据。
  - 不要引入 @radix-ui/react-accordion 或其他第三方手风琴库。
  - 如需扩展行为（多展开、替换图标等），在组件源码中修改。
