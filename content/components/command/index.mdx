Command 命令面板，提供搜索过滤和快捷指令选择，常用于 ⌘K 面板。

<ComponentShowcase id="command" path="command/basic" />

```tsx
import { Command, CommandGroup, CommandInput, CommandItem, CommandList } from "@/components/ui/command"
export default function Demo() {
  return (
    <Command className="w-64 rounded border">
      <CommandInput placeholder="搜索命令..." />
      <CommandList><CommandGroup heading="操作"><CommandItem>新建文件</CommandItem><CommandItem>打开文件夹</CommandItem></CommandGroup></CommandList>
    </Command>
  )
}
```

子组件：`Command`、`CommandInput`、`CommandList`、`CommandGroup`、`CommandItem`、`CommandEmpty`。
