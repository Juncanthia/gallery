Timeline 垂直展示时间流信息，按时间顺序排列事件节点。

<ComponentShowcase id="timeline" path="timeline/basic" />

```tsx
import { Timeline } from "@/components/ui/timeline"
export default function Demo() {
  return (
    <Timeline
      items={[
        { label: "2024-01-01", children: "项目启动" },
        { label: "2024-02-15", children: "第一阶段上线" },
        { label: "2024-03-30", color: "gray", children: "等待第二阶段" },
      ]}
      className="max-w-sm"
    />
  )
}
```

| 属性 | 说明 | 类型 |
| --- | --- | --- |
| items | 时间线项 | `{ label?, children, color? }[]` |
