ContextMenu 右键弹出菜单，基于 Radix ContextMenu 实现。

<ComponentShowcase id="context-menu" path="context-menu/basic" />

```tsx
import { ContextMenu, ContextMenuContent, ContextMenuItem, ContextMenuSeparator, ContextMenuTrigger } from "@/components/ui/context-menu"
export default function Demo() {
  return (
    <ContextMenu>
      <ContextMenuTrigger className="flex h-32 w-64 items-center justify-center rounded border border-dashed text-sm text-muted-foreground">右键点击此处</ContextMenuTrigger>
      <ContextMenuContent><ContextMenuItem>复制</ContextMenuItem><ContextMenuSeparator /><ContextMenuItem>删除</ContextMenuItem></ContextMenuContent>
    </ContextMenu>
  )
}
```
