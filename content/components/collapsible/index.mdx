Collapsible 折叠容器，点击触发器展开/收起内容，基于 Radix Collapsible 实现。

<ComponentShowcase id="collapsible" path="collapsible/basic" />

```tsx
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from "@/components/ui/collapsible"
import { Button } from "@/components/ui/button"
import { ChevronDown } from "lucide-react"

export default function Demo() {
  return (
    <Collapsible className="w-full max-w-sm space-y-2">
      <div className="flex items-center justify-between">
        <h4 className="text-sm font-semibold">可折叠区域</h4>
        <CollapsibleTrigger asChild>
          <Button variant="text" size="small" icon={<ChevronDown className="size-4" />} />
        </CollapsibleTrigger>
      </div>
      <CollapsibleContent className="rounded border px-4 py-3 text-sm text-muted-foreground">
        折叠面板的内容，展开后可见。
      </CollapsibleContent>
    </Collapsible>
  )
}
```

| 属性 | 说明 | 类型 | 默认值 |

| onOpenChange | 展开回调 | `(open: boolean) => void` | - |
